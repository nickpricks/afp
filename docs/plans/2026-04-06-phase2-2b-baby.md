// src/modules/baby/utils.ts
/** Computes display age from date of birth. */
export const computeAge = (dob: string): string => {
  const birth = new Date(dob);
  const now = new Date();
  const months = (now.getFullYear() - birth.getFullYear()) * 12 + (now.getMonth() - birth.getMonth());
  if (months < 1) return 'Newborn';
  if (months < 24) return `${months} months`;
  return `${Math.floor(months / 12)} years`;
};
```

import { describe, it, expect } from 'vitest';
import { computeAge } from '../utils';

describe('computeAge', () => {
  it('returns "Newborn" for recent birth', () => {
    const today = new Date();
    const dob = today.toISOString().split('T')[0];
    expect(computeAge(dob)).toBe('Newborn');
  });

  it('returns months for < 2 years', () => {
    const d = new Date();
    d.setMonth(d.getMonth() - 8);
    expect(computeAge(d.toISOString().split('T')[0])).toBe('8 months');
  });

  it('returns years for >= 2 years', () => {
    const d = new Date();
    d.setFullYear(d.getFullYear() - 3);
    expect(computeAge(d.toISOString().split('T')[0])).toBe('3 years');
  });
});
```
