```typescript
// src/modules/body/hooks/useBodyConfig.ts
import { useState, useEffect, useCallback } from 'react';
import { useAuth } from '@/shared/auth/useAuth';
import { createAdapter } from '@/shared/storage/create-adapter';
import { DbSubcollection, DbDoc } from '@/constants/db';
import type { BodyConfig } from '../types';

/** Hook for reading and saving body module configuration. */
export const useBodyConfig = () => {
  const { firebaseUser } = useAuth();
  const [config, setConfig] = useState<BodyConfig | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (!firebaseUser) return;
    const adapter = createAdapter(`users/${firebaseUser.uid}/${DbSubcollection.BodyConfig}`);
    const unsub = adapter.onSnapshot<BodyConfig>(
      DbDoc.Main,
      (data) => {
        setConfig(data);
        setLoading(false);
      },
      () => setLoading(false),
    );
    return unsub;
  }, [firebaseUser]);

  const saveConfig = useCallback(
    async (newConfig: BodyConfig) => {
      if (!firebaseUser) return;
      const adapter = createAdapter(`users/${firebaseUser.uid}/${DbSubcollection.BodyConfig}`);
      await adapter.save(DbDoc.Main, newConfig);
      setConfig(newConfig);
    },
    [firebaseUser],
  );

  return { config, isConfigured: config !== null, loading, saveConfig };
};
```
